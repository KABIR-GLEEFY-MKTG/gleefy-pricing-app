const express = require("express"); const path = require("path"); const app = express(); app.use(express.json()); app.use(express.static(path.join(__dirname))); app.get("/", (req, res) => { res.sendFile(path.join(__dirname, "index.html"), (err) => { if (err) { res.status(500).send("Error: " + err.message + " | Dir: " + __dirname); } }); }); const PORT = process.env.PORT || 3001; app.listen(PORT, () => console.log("Gleefy running on port " + PORT));
